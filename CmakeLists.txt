cmake_minimum_required(VERSION 3.16...3.30)
set(CMAKE_POLICY_VERSION_MINIMUM 3.5)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

project(SteamUI VERSION 0.1 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Qt6 REQUIRED COMPONENTS Gui Qml Quick QuickControls2 Concurrent Core LinguistTools Widgets)
qt_standard_project_setup()

include(FetchContent)

find_package(CURL)

set(JSON_BuildTests OFF CACHE INTERNAL "")
FetchContent_Declare(json URL https://github.com/nlohmann/json/releases/download/v3.11.2/json.tar.xz)
FetchContent_MakeAvailable(json)

FetchContent_Declare(cpr 
    GIT_REPOSITORY https://github.com/libcpr/cpr.git 
    GIT_TAG 1.10.x
)

if(CURL_FOUND)
    set(CPR_FORCE_USE_SYSTEM_CURL ON CACHE BOOL "" FORCE)
endif()

FetchContent_MakeAvailable(cpr)

add_executable(Steam_Coverlay
    include/steamgrid.h
    source/main.cpp
    source/steamgrid.cpp
)

if(WIN32)
    set(app_icon_resource "${CMAKE_CURRENT_SOURCE_DIR}/resources/logo.ico")
    target_sources(Steam_Coverlay PRIVATE 
        ${app_icon_resource}
        resources/resources.rc
    )
    set_target_properties(Steam_Coverlay PROPERTIES WIN32_EXECUTABLE TRUE)
endif()

target_include_directories(Steam_Coverlay PRIVATE include)

qt_add_qml_module(Steam_Coverlay
    URI SteamApp
    VERSION 1.0
    QML_FILES 
        ui/Main.qml
        ui/Theme.qml
        ui/CsButton.qml
        ui/Settings.qml
        ui/Progress.qml
        ui/Fonts.qml
        ui/CsComponent.qml
        ui/ResultsGrid.qml
        ui/SearchBar.qml
    RESOURCES
        resources/logo.ico
        resources/set.svg
        resources/loop.svg
        resources/messages_pl.json
        resources/messages_ja.json
        resources/messages_en.json
        resources/directory.svg
        fonts/Inter.ttf
        fonts/Inter-Bold.ttf  
        resources/skeleton/1.png
        resources/skeleton/2.png
        resources/skeleton/3.png
        resources/skeleton/4.png
        resources/skeleton/5.png
        resources/skeleton/6.png
        resources/skeleton/7.png
        resources/skeleton/8.png
)

target_link_libraries(Steam_Coverlay PRIVATE 
    Qt6::Gui 
    Qt6::Qml 
    Qt6::Quick
    Qt6::QuickControls2
    Qt6::Concurrent 
    Qt6::Core
    Qt6::Widgets
    cpr::cpr
    nlohmann_json::nlohmann_json
)

set(TS_FILES
    translations/app_pl.ts
    translations/app_en.ts
    translations/app_ja.ts
)

qt_add_translations(Steam_Coverlay   
    TS_FILES ${TS_FILES}
    LUPDATE_OPTIONS "-extensions" "qml,cpp,h"
    SOURCES
        ui/Main.qml
        ui/SearchBar.qml
        ui/ResultsGrid.qml
        ui/Settings.qml
        ui/Progress.qml
        ui/CsComponent.qml
        ui/CsButton.qml
        source/steamgrid.cpp
        include/steamgrid.h
)
if(NOT WIN32)
    # 1. Generowanie pliku .desktop w folderze budowania
    configure_file(linux/steamui.desktop.in ${CMAKE_CURRENT_BINARY_DIR}/steamui.desktop @ONLY)

    # 2. Definiowanie gdzie mają trafić pliki podczas instalacji (wymagane dla AppImage)
    install(TARGETS Steam_Coverlay DESTINATION bin)
    install(FILES ${CMAKE_CURRENT_BINARY_DIR}/steamui.desktop DESTINATION share/applications)
    
    # Jako ikonę systemową instalujemy plik set.svg i zmieniamy jego nazwę na steamui.svg
    install(FILES resources/set.svg DESTINATION share/icons/hicolor/scalable/apps RENAME steamui.svg)
endif()
